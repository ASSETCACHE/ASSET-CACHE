#!/usr/bin/env bash
# Push script to create the repo (if needed) and push the current directory to GitHub.
# Usage:
# 1) Place this script in the same folder as index.html, README.md, .gitignore
# 2) chmod +x push.sh
# 3) ./push.sh
#
# Requires: git and (optionally) GitHub CLI `gh` configured for your account.
set -euo pipefail

REPO_OWNER="ASSETCACHE"
REPO_NAME="Asset-Cache"
BRANCH="main"
COMMIT_MSG="initial Commit"
REMOTE_SSH="git@github.com:${REPO_OWNER}/${REPO_NAME}.git"
REMOTE_HTTPS="https://github.com/${REPO_OWNER}/${REPO_NAME}.git"

# If gh is available, try to create the repo if it doesn't exist
if command -v gh >/dev/null 2>&1; then
if ! gh repo view "${REPO_OWNER}/${REPO_NAME}" >/dev/null 2>&1; then
echo "Creating GitHub repo ${REPO_OWNER}/${REPO_NAME} (public)..."
gh repo create "${REPO_OWNER}/${REPO_NAME}" --public --description "Assets Secured Shared Equity Trust" --confirm
else
echo "GitHub repo ${REPO_OWNER}/${REPO_NAME} already exists."
fi
else
echo "gh CLI not found. Make sure the remote repo exists at ${REMOTE_HTTPS} or install and authenticate gh."
fi

# Initialize local repo if needed
if [ ! -d .git ]; then
echo "Initializing git repository and setting branch to ${BRANCH}..."
git init -b "${BRANCH}"
fi

# Ensure there are files to commit
git add --all
if git diff --cached --quiet; then
echo "No changes staged for commit. Skipping commit step."
else
git commit -m "${COMMIT_MSG}"
fi

# Add/set remote
if git remote | grep -q "^origin$"; then
git remote set-url origin "${REMOTE_SSH}" 2>/dev/null || git remote set-url origin "${REMOTE_HTTPS}"
else
git remote add origin "${REMOTE_SSH}" 2>/dev/null || git remote add origin "${REMOTE_HTTPS}"
fi

# Push. If remote has unrelated history, hint to the user.
echo "Pushing to origin ${BRANCH}..."
if git ls-remote --exit-code --heads origin "${BRANCH}" >/dev/null 2>&1; then
# Branch exists on remote; do a normal push (may fail if histories diverge)
if git push -u origin "${BRANCH}"; then
echo "Pushed to origin/${BRANCH}."
else
echo "Push failed due to diverging history. To force overwrite remote branch (CAUTION: will replace remote):"
echo " git push -u origin ${BRANCH} --force"
echo "Or merge remote changes first:"
echo " git fetch origin"
echo " git merge origin/${BRANCH} --allow-unrelated-histories"
fi
else
# Branch doesn't exist on remote, push normally
git push -u origin "${BRANCH}"
echo "Pushed to origin/${BRANCH}."
fi

echo "Done."
